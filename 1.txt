#include<stdio.h>
#include <stdint.h>

#define LEN 10
int len = LEN;
int out[LEN];

void set_zero(int start)
{
    for(int i=start;i<len;i++){
        out[i] = 0;
    }
    return;
}
int set_data(int data, int start)
{
    int i;
    for(i=0;i<data && i + start < len;i++){
        out[i+start]=1;
    }
    if(i==data){
        return i + start;
    }
    return -1;
}

int dfs(int* in,int in_cnt, int n, int start)
{
    int ret, i;
    if(n == in_cnt){
        for(int i=0;i<len;i++){
            printf("%d ", out[i]);
        }
        printf("\n");
        return 0;
    }
    for(i=start;i< len;i++){
        if(n!=0 && i == start){
            continue;
        }
        set_zero(start);

        ret = set_data(in[n], i);
        if(ret == -1){
            break;
        }
        for(int j=in[n]+ i;j <= len;j++){
            set_zero(j);
            if(!dfs(in, in_cnt, n+1, j)){
                break;
            }
        }
    }
    return 0;
}
int main()
{
    int in[]={3,3,2};
    int in_len = sizeof(in)/sizeof(in[0]);
    int sum=0;
    for(int i=0;i<in_len;i++){
        sum += in[i];
    }
    if(sum + in_len - 1 > len){
        printf("input error.\n");
        return -1;
    }
    dfs(in, in_len, 0, 0);
    return 0;
}
